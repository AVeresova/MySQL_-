1. Общее текстовое описание.
БД "Auto" предназначена для учета информации по закупкам, 
хранению и реализации автомобилей дилерским автоцентром.


2. Скрипты создания БД.
CREATE SCHEMA `auto` DEFAULT CHARACTER SET utf8


CREATE TABLE IF NOT EXISTS `auto`.`Car_dealerships` (
  `id` INT(10) UNSIGNED NOT NULL AUTO_INCREMENT,
  `name` VARCHAR(45) NOT NULL,
  `address` VARCHAR(145) NOT NULL,
  `phone` BIGINT(12) UNSIGNED NOT NULL,
  `region` VARCHAR(45) NOT NULL,
  PRIMARY KEY (`id`),
  UNIQUE INDEX `phone_UNIQUE` (`phone` ASC) VISIBLE,
  UNIQUE INDEX `address_UNIQUE` (`address` ASC) VISIBLE)
ENGINE = InnoDB
DEFAULT CHARACTER SET = utf8;

CREATE TABLE IF NOT EXISTS `auto`.`Suppliers` (
  `id` INT(10) UNSIGNED NOT NULL AUTO_INCREMENT,
  `name` VARCHAR(145) NOT NULL,
  `product_type` VARCHAR(45) NOT NULL,
  `contract` VARCHAR(45) NOT NULL,
  `date` DATE NOT NULL,
  PRIMARY KEY (`id`))
ENGINE = InnoDB
DEFAULT CHARACTER SET = utf8;

CREATE TABLE IF NOT EXISTS `auto`.`Сustomer` (
  `id` INT(10) UNSIGNED NOT NULL AUTO_INCREMENT,
  `name` VARCHAR(245) NOT NULL COMMENT 'Фамилия Имя Отчество, либо наименование юридического лица',
  `type` ENUM("физическое лицо", "юридическое лицо") NOT NULL COMMENT 'Физическое лицо/юридическое лицо',
  `e-mail` VARCHAR(145) NOT NULL,
  `phone` BIGINT(12) UNSIGNED NOT NULL,
  PRIMARY KEY (`id`),
  UNIQUE INDEX `e-mail_UNIQUE` (`e-mail` ASC) VISIBLE,
  UNIQUE INDEX `phone_UNIQUE` (`phone` ASC) VISIBLE)
ENGINE = InnoDB
DEFAULT CHARACTER SET = utf8;

CREATE TABLE IF NOT EXISTS `auto`.`Auto` (
  `id` INT(11) NOT NULL AUTO_INCREMENT,
  `brand` VARCHAR(45) NOT NULL,
  `model_group` VARCHAR(45) NOT NULL,
  `model` VARCHAR(45) NOT NULL,
  `Suppliers_id` INT(10) UNSIGNED NOT NULL,
  `Warehouse_id` INT(11) NOT NULL,
  PRIMARY KEY (`id`),
  INDEX `fk_Auto_Suppliers_idx` (`Suppliers_id` ASC) VISIBLE,
  INDEX `fk_Auto_Warehouse1_idx` (`Warehouse_id` ASC) VISIBLE,
  CONSTRAINT `fk_Auto_Suppliers`
    FOREIGN KEY (`Suppliers_id`)
    REFERENCES `auto`.`Suppliers` (`id`)
    ON DELETE NO ACTION
    ON UPDATE NO ACTION,
  CONSTRAINT `fk_Auto_Warehouse1`
    FOREIGN KEY (`Warehouse_id`)
    REFERENCES `auto`.`Warehouse` (`id`)
    ON DELETE NO ACTION
    ON UPDATE NO ACTION)
ENGINE = InnoDB
DEFAULT CHARACTER SET = utf8;

CREATE TABLE IF NOT EXISTS `auto`.`Warehouse` (
  `id` INT(11) NOT NULL AUTO_INCREMENT,
  `name` VARCHAR(45) NOT NULL,
  PRIMARY KEY (`id`))
ENGINE = InnoDB
DEFAULT CHARACTER SET = utf8;

CREATE TABLE IF NOT EXISTS `auto`.`Auto_parts` (
  `id` INT(11) NOT NULL AUTO_INCREMENT,
  `number` VARCHAR(45) NOT NULL,
  `name` VARCHAR(145) NOT NULL,
  `Suppliers_id` INT(10) UNSIGNED NOT NULL,
  `Warehouse_id` INT(11) NOT NULL,
  PRIMARY KEY (`id`),
  INDEX `fk_Auto_parts_Suppliers1_idx` (`Suppliers_id` ASC) VISIBLE,
  UNIQUE INDEX `number_UNIQUE` (`number` ASC) VISIBLE,
  INDEX `fk_Auto_parts_Warehouse1_idx` (`Warehouse_id` ASC) VISIBLE,
  CONSTRAINT `fk_Auto_parts_Suppliers1`
    FOREIGN KEY (`Suppliers_id`)
    REFERENCES `auto`.`Suppliers` (`id`)
    ON DELETE NO ACTION
    ON UPDATE NO ACTION,
  CONSTRAINT `fk_Auto_parts_Warehouse1`
    FOREIGN KEY (`Warehouse_id`)
    REFERENCES `auto`.`Warehouse` (`id`)
    ON DELETE NO ACTION
    ON UPDATE NO ACTION)
ENGINE = InnoDB
DEFAULT CHARACTER SET = utf8;

CREATE TABLE IF NOT EXISTS `auto`.`Register_Auto` (
  `id` INT(11) NOT NULL AUTO_INCREMENT,
  `Auto_id` INT(11) NOT NULL,
  `quantity` INT(10) UNSIGNED NOT NULL COMMENT 'Количество',
  `price` FLOAT(11) NOT NULL COMMENT 'Цена',
  `cost` FLOAT(11) NOT NULL COMMENT 'Стоимость',
  `date` DATE NOT NULL,
  `Car_dealerships_id` INT(10) UNSIGNED NOT NULL,
  `VIN` VARCHAR(17) NOT NULL,
  INDEX `fk_Register_Auto_Auto1_idx` (`Auto_id` ASC) VISIBLE,
  PRIMARY KEY (`id`),
  INDEX `fk_Register_Auto_Car_dealerships1_idx` (`Car_dealerships_id` ASC) VISIBLE,
  UNIQUE INDEX `VIN_UNIQUE` (`VIN` ASC) VISIBLE,
  CONSTRAINT `fk_Register_Auto_Auto1`
    FOREIGN KEY (`Auto_id`)
    REFERENCES `auto`.`Auto` (`id`)
    ON DELETE NO ACTION
    ON UPDATE NO ACTION,
  CONSTRAINT `fk_Register_Auto_Car_dealerships1`
    FOREIGN KEY (`Car_dealerships_id`)
    REFERENCES `auto`.`Car_dealerships` (`id`)
    ON DELETE NO ACTION
    ON UPDATE NO ACTION)
ENGINE = InnoDB
DEFAULT CHARACTER SET = utf8;

CREATE TABLE IF NOT EXISTS `auto`.`Register_Auto_parts` (
  `id` INT(10) UNSIGNED NOT NULL AUTO_INCREMENT,
  `Auto_parts_id` INT(11) NOT NULL,
  `quantity` INT(10) UNSIGNED NOT NULL,
  `price` FLOAT(10) UNSIGNED NOT NULL,
  `cost` FLOAT(10) UNSIGNED NOT NULL,
  `Car_dealerships_id` INT(10) UNSIGNED NOT NULL,
  `date` DATE NOT NULL,
  INDEX `fk_Register_Auto_parts_Auto_parts1_idx` (`Auto_parts_id` ASC) VISIBLE,
  PRIMARY KEY (`id`),
  INDEX `fk_Register_Auto_parts_Car_dealerships1_idx` (`Car_dealerships_id` ASC) VISIBLE,
  CONSTRAINT `fk_Register_Auto_parts_Auto_parts1`
    FOREIGN KEY (`Auto_parts_id`)
    REFERENCES `auto`.`Auto_parts` (`id`)
    ON DELETE NO ACTION
    ON UPDATE NO ACTION,
  CONSTRAINT `fk_Register_Auto_parts_Car_dealerships1`
    FOREIGN KEY (`Car_dealerships_id`)
    REFERENCES `auto`.`Car_dealerships` (`id`)
    ON DELETE NO ACTION
    ON UPDATE NO ACTION)
ENGINE = InnoDB
DEFAULT CHARACTER SET = utf8;

CREATE TABLE IF NOT EXISTS `auto`.`Managers` (
  `id` INT(11) NOT NULL AUTO_INCREMENT,
  `name` VARCHAR(145) NOT NULL,
  `position` VARCHAR(45) NOT NULL,
  `date_of_employment` DATE NOT NULL,
  PRIMARY KEY (`id`))
ENGINE = InnoDB
DEFAULT CHARACTER SET = utf8;

CREATE TABLE IF NOT EXISTS `auto`.`Sale_Auto` (
  `id` INT(10) UNSIGNED NOT NULL AUTO_INCREMENT,
  `Auto_id` INT(11) NOT NULL,
  `Managers_id` INT(11) NOT NULL,
  `Сustomer_id` INT(10) UNSIGNED NOT NULL,
  `price` FLOAT(10) UNSIGNED NOT NULL,
  `Register_Auto_id` INT(11) NOT NULL,
  `date` DATE NOT NULL,
  PRIMARY KEY (`id`),
  INDEX `fk_Sale_Auto_Auto1_idx` (`Auto_id` ASC) VISIBLE,
  INDEX `fk_Sale_Auto_Managers1_idx` (`Managers_id` ASC) VISIBLE,
  INDEX `fk_Sale_Auto_Сustomer1_idx` (`Сustomer_id` ASC) VISIBLE,
  INDEX `fk_Sale_Auto_Register_Auto1_idx` (`Register_Auto_id` ASC) VISIBLE,
  CONSTRAINT `fk_Sale_Auto_Auto1`
    FOREIGN KEY (`Auto_id`)
    REFERENCES `auto`.`Auto` (`id`)
    ON DELETE NO ACTION
    ON UPDATE NO ACTION,
  CONSTRAINT `fk_Sale_Auto_Managers1`
    FOREIGN KEY (`Managers_id`)
    REFERENCES `auto`.`Managers` (`id`)
    ON DELETE NO ACTION
    ON UPDATE NO ACTION,
  CONSTRAINT `fk_Sale_Auto_Сustomer1`
    FOREIGN KEY (`Сustomer_id`)
    REFERENCES `auto`.`Сustomer` (`id`)
    ON DELETE NO ACTION
    ON UPDATE NO ACTION,
  CONSTRAINT `fk_Sale_Auto_Register_Auto1`
    FOREIGN KEY (`Register_Auto_id`)
    REFERENCES `auto`.`Register_Auto` (`id`)
    ON DELETE NO ACTION
    ON UPDATE NO ACTION)
ENGINE = InnoDB
DEFAULT CHARACTER SET = utf8;
